#include <iostream>
#include <span>

#include "String.h"
#include "save.h"


using namespace std;
extern bool 관찰;

//void print(int[]);	// 인자 int[]는 int*로 collapse된다. (정보가 줄어든다)
void print(span<int>);	// 인자는 contiguous메모리면 된다.

//-----------
int main()
//-----------
{
	int a[10]{ 1,2,3,4,5 };
	print(a); // a의 원소를 화면 출력하는 함수

	// 기존 인자는 배열의 길이같은걸 모르기 때문에 배열을 출력하는 행동등을 하려면 애를 먹었다.
	// 하지만 컴파일러는 a의 시작과 끝, 사이즈 등등을 알기 때문에 span 포장술로 인자 넘겨주면 인자 받은
	// 함수에서도 배열의 사이즈등을 알 수 있다.

	save("소스.cpp");
}

void print(span<int> a) {
	for (int t : a)
		cout << t << endl;
}
