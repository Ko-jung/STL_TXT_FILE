#include <iostream>
#include <fstream>
#include <algorithm>
#include "save.h"

using namespace std;

// [문제] 소스.cpp에 있는 문자를 모두 대문자로 바꾸어라.
// 결과는 "소스대문자.cpp"에 저장하라.

//-----------
int main()
//-----------
{
	ifstream in{ "소스.cpp" };
	if (!in) {
		return 0;
	}
	ofstream out{ "소스대문자.cpp" };

	// transform(입력 파일을, 어디까지, 출력 파일로, 대문자로 바꿔서(함수))
	transform(istreambuf_iterator<char>{in}, {},
		ostreambuf_iterator<char>{cout}, [](char c) {
			return c = toupper(c);
		});
	
	save("소스.cpp");
}

#include <iostream>
#include <fstream>
#include <algorithm>
#include "save.h"

using namespace std;

// [문제] 숫자를 *로 바꿔보겠다.

//-----------
int main()
//-----------
{
	ifstream in{ "소스.cpp" };
	if (!in) {
		return 0;
	}
	ofstream out{ "소스대문자.cpp" };

	// transform(입력 파일을, 어디까지, 출력 파일로, 대문자로 바꿔서(함수))
	transform(istreambuf_iterator<char>{in}, {},
		ostreambuf_iterator<char>{cout}, [](char c) {
			if (isdigit(c)) {
				c = '*';
			}
			return c;
		});
	
	save("소스.cpp");
}
