#include <iostream>

#include <vector>

#include "String.h"

using namespace std;

extern bool 관찰;

//-----------
int main()
//-----------
{
	//소수(prime number)를 출력하는 프로그램을 작성한다
	// 10000부터 20000까지 정수 중에서 소수만을 출력하라

	auto comp = [](int n) {
		for (int i = 2; i < sqrt(n); i++)
			if (n % i == 0)
				return false;
		return true;
	};
	

	//무한개를 만들어서 100개를 취하는게 아님을 알 수 있음
	for (int n : views::iota(10000) /* | views::drop(10000)*/
									| views::take(10001)
									| views::filter(comp))
		cout << n << ' ';
	save("소스.cpp");
}
