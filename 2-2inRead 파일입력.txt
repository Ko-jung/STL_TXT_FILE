#include <iostream>
#include <fstream>
#include <array>
#include <format>
#include <chrono>

#include "save.h"

using namespace std;

// [문제] 파일 "int 천 개를 메모리 그대로"에는 int 천 개가 정확하게 저장되어 있다.
// 4'000바이트를 binary mode로 write 함수로 기록하였다.
// 읽어서 화면에 출력하라.

//-----------
int main()
//-----------
{
	array<int, 1'000> a;

	ifstream in{ "int 천 개를 메모리 그대로", ios::binary };
	if (!in) {
		cout << "파일을 찾을 수 없습니다.\n";
		return 0;
	}

	in.read((char*)&a, sizeof(int) * a.size());

	for (int d : a) {
		cout << format("{:8}", d);
		this_thread::sleep_for(100ns);
	}

	save("소스.cpp");
}
